## 练习 1：创建维度模型（星型模式）

### 1.1 创建维度表

```sql
-- ========== 时间维度表 ==========
CREATE TABLE dim_time (
    time_id     INT PRIMARY KEY,
    full_date   DATE NOT NULL,
    year        INT NOT NULL,
    quarter     INT NOT NULL,      -- 1, 2, 3, 4
    month       INT NOT NULL,      -- 1-12
    day         INT NOT NULL,      -- 1-31
    weekday     VARCHAR(10),       -- 'Monday', 'Tuesday'...
    is_weekend  INT DEFAULT 0      -- 0=工作日, 1=周末
);

-- ========== 产品维度表 ==========
CREATE TABLE dim_product (
    product_id  INT PRIMARY KEY,
    name        VARCHAR(100) NOT NULL,
    category    VARCHAR(50),      -- '电子产品', '食品', '服装'
    subcategory VARCHAR(50),      -- '手机', '电脑', '平板'
    brand       VARCHAR(50),
    unit_price  DECIMAL(10,2)
);

-- ========== 门店维度表 ==========
CREATE TABLE dim_store (
    store_id    INT PRIMARY KEY,
    name        VARCHAR(100) NOT NULL,
    city        VARCHAR(50),
    region      VARCHAR(50),      -- '华东', '华北', '华南'
    country     VARCHAR(50),
    store_type  VARCHAR(20)       -- '旗舰店', '标准店', '社区店'
);

-- ========== 客户维度表 ==========
CREATE TABLE dim_customer (
    customer_id INT PRIMARY KEY,
    name        VARCHAR(100),
    gender      VARCHAR(10),
    age_group   VARCHAR(20),      -- '18-25', '26-35', '36-50', '50+'
    city        VARCHAR(50),
    member_level VARCHAR(20)      -- '普通', '银卡', '金卡', '钻石'
);

-- ========== 销售事实表（核心） ==========
CREATE TABLE fact_sales (
    sale_id         INT PRIMARY KEY,
    time_id         INT NOT NULL REFERENCES dim_time(time_id),
    product_id      INT NOT NULL REFERENCES dim_product(product_id),
    store_id        INT NOT NULL REFERENCES dim_store(store_id),
    customer_id     INT NOT NULL REFERENCES dim_customer(customer_id),
    quantity        INT NOT NULL,          -- 销售数量
    amount          DECIMAL(12,2) NOT NULL, -- 销售金额
    discount        DECIMAL(5,2) DEFAULT 0, -- 折扣金额
    cost            DECIMAL(12,2)         -- 成本（用于计算利润）
);
```

## 练习 2：基础分析查询

### 2.1 各产品总销售额排名

```sql
SELECT 
    p.name AS 产品,
    SUM(f.amount) AS 总销售额,
    SUM(f.quantity) AS 总销量,
    COUNT(DISTINCT f.customer_id) AS 购买客户数
FROM fact_sales f
JOIN dim_product p ON f.product_id = p.product_id
GROUP BY p.name
ORDER BY 总销售额 DESC;
```

**你的任务**：执行查询，观察结果，回答：
- 哪个产品销售额最高？
- 销量和销售额是否成正比？为什么？

### 2.2 各城市销售额对比

```sql
SELECT 
    s.city AS 城市,
    s.region AS 区域,
    SUM(f.amount) AS 销售额,
    COUNT(*) AS 订单数,
    ROUND(AVG(f.amount), 2) AS 客单价
FROM fact_sales f
JOIN dim_store s ON f.store_id = s.store_id
GROUP BY s.city, s.region
ORDER BY 销售额 DESC;
```

**你的任务**：加上 `ROUND(SUM(f.amount) / SUM(SUM(f.amount)) OVER() * 100, 2)` 计算各城市占比。

---

### 2.3 各品类在各区域的销售

```sql
SELECT 
    p.category AS 品类,
    s.region AS 区域,
    SUM(f.amount) AS 销售额
FROM fact_sales f
JOIN dim_product p ON f.product_id = p.product_id
JOIN dim_store s ON f.store_id = s.store_id
GROUP BY p.category, s.region
ORDER BY 品类, 销售额 DESC;
```

**你的任务**：用 `PIVOT` 或 `CASE WHEN` 把上面的结果转成交叉表（品类为行，区域为列）。

## 练习 3：OLAP 操作实战

### 3.1 Drill-Down（下钻）：从品类到子品类

```sql
-- Roll-Up 到品类
SELECT p.category, SUM(f.amount) 
FROM fact_sales f JOIN dim_product p ON f.product_id = p.product_id
GROUP BY p.category;

-- Drill-Down 到子品类 
SELECT p.category, p.subcategory, SUM(f.amount)
FROM fact_sales f JOIN dim_product p ON f.product_id = p.product_id
GROUP BY p.category, p.subcategory
ORDER BY p.category;
```

### 3.2 Roll-Up（上卷）：从城市到区域

```sql
-- 城市级别
SELECT s.city, SUM(f.amount) FROM fact_sales f
JOIN dim_store s ON f.store_id = s.store_id
GROUP BY s.city;

-- 上卷到区域级别
SELECT s.region, SUM(f.amount) FROM fact_sales f
JOIN dim_store s ON f.store_id = s.store_id
GROUP BY s.region;
```

### 3.3 Slice（切片）：只看"电子产品"

```sql
SELECT 
    p.name,
    s.city,
    SUM(f.amount) 
FROM fact_sales f
JOIN dim_product p ON f.product_id = p.product_id
JOIN dim_store s ON f.store_id = s.store_id
WHERE p.category = '电子产品'      -- ← 切片条件
GROUP BY p.name, s.city;
```

### 3.4 Dice（切块）：电子产品 + 华东区 + 1月

```sql
SELECT 
    p.name,
    s.city,
    t.full_date,
    SUM(f.amount)
FROM fact_sales f
JOIN dim_product p ON f.product_id = p.product_id
JOIN dim_store s ON f.store_id = s.store_id
JOIN dim_time t ON f.time_id = t.time_id
WHERE p.category = '电子产品'       -- 条件1
  AND s.region = '华东'             -- 条件2
  AND t.month = 1                   -- 条件3
GROUP BY p.name, s.city, t.full_date;
```

## 练习 4：窗口函数进阶

### 4.1 移动平均（7天）

```sql
SELECT 
    t.full_date,
    SUM(f.amount) AS 日销售额,
    ROUND(AVG(SUM(f.amount)) OVER (ORDER BY t.full_date ROWS BETWEEN 6 PRECEDING AND CURRENT ROW), 2) AS 七日移动平均
FROM fact_sales f
JOIN dim_time t ON f.time_id = t.time_id
GROUP BY t.full_date
ORDER BY t.full_date;
```

### 4.2 ABC 分类（帕累托分析）

```sql
WITH product_sales AS (
    SELECT 
        p.name,
        SUM(f.amount) AS total,
        SUM(SUM(f.amount)) OVER () AS grand_total
    FROM fact_sales f 
    JOIN dim_product p ON f.product_id = p.product_id
    GROUP BY p.name
),
ranked AS (
    SELECT 
        name,
        total,
        ROUND(total / grand_total * 100, 2) AS pct,
        ROUND(SUM(total) OVER (ORDER BY total DESC) / grand_total * 100, 2) AS cumulative_pct
    FROM product_sales
)
SELECT 
    name,
    total,
    pct,
    cumulative_pct,
    CASE 
        WHEN cumulative_pct <= 70 THEN 'A类（重点）'
        WHEN cumulative_pct <= 90 THEN 'B类（一般）'
        ELSE 'C类（长尾）'
    END AS 分类
FROM ranked
ORDER BY total DESC;
```

```sql
-- Q1: 哪个区域增长最快（假设有去年数据）
SELECT 
    s.region,
    SUM(CASE WHEN t.year = 2024 THEN f.amount ELSE 0 END) AS 今年,
    SUM(CASE WHEN t.year = 2023 THEN f.amount ELSE 0 END) AS 去年,
    ROUND((SUM(CASE WHEN t.year = 2024 THEN f.amount ELSE 0 END) 
         - SUM(CASE WHEN t.year = 2023 THEN f.amount ELSE 0 END))
         / NULLIF(SUM(CASE WHEN t.year = 2023 THEN f.amount ELSE 0 END), 0) * 100, 2) AS 增长率
FROM fact_sales f
JOIN dim_store s ON f.store_id = s.store_id
JOIN dim_time t ON f.time_id = t.time_id
GROUP BY s.region;

-- Q2: 周末 vs 工作日
SELECT 
    t.is_weekend,
    CASE WHEN t.is_weekend = 1 THEN '周末' ELSE '工作日' END AS 日期类型,
    SUM(f.amount) AS 总销售额,
    SUM(f.quantity) AS 总销量,
    ROUND(AVG(f.amount), 2) AS 平均客单价,
    COUNT(*) AS 订单数
FROM fact_sales f
JOIN dim_time t ON f.time_id = t.time_id
GROUP BY t.is_weekend;

-- Q3: 滞销品（销量倒数 20%）
WITH ranked AS (
    SELECT 
        p.name,
        SUM(f.quantity) AS total_qty,
        PERCENT_RANK() OVER (ORDER BY SUM(f.quantity)) AS pct_rank
    FROM fact_sales f
    JOIN dim_product p ON f.product_id = p.product_id
    GROUP BY p.name
)
SELECT name, total_qty
FROM ranked
WHERE pct_rank <= 0.2
ORDER BY total_qty;

-- Q4: 核心客户画像（贡献 80% 销售额的客户群）
WITH customer_revenue AS (
    SELECT 
        c.customer_id,
        c.gender,
        c.age_group,
        c.city,
        c.member_level,
        SUM(f.amount) AS revenue,
        SUM(SUM(f.amount)) OVER () AS total_rev
    FROM fact_sales f
    JOIN dim_customer c ON f.customer_id = c.customer_id
    GROUP BY c.customer_id, c.gender, c.age_group, c.city, c.member_level
),
ranked AS (
    SELECT *,
        SUM(revenue) OVER (ORDER BY revenue DESC) / total_rev AS cumulative_pct
    FROM customer_revenue
)
SELECT gender, age_group, member_level, COUNT(*) AS 人数, SUM(revenue) AS 贡献额
FROM ranked
WHERE cumulative_pct <= 0.8
GROUP BY gender, age_group, member_level
ORDER BY 贡献额 DESC;
```
